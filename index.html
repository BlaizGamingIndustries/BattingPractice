<script>

    const canvas = document.getElementById('gameCanvas');

    const ctx = canvas.getContext('2d');

    const hitsVal = document.getElementById('hits-val');

    const streakVal = document.getElementById('streak-val');

    const pitchFeedback = document.getElementById('pitch-feedback');

    const swingBtn = document.getElementById('swing-btn');


    function resizeCanvas() {

        const container = document.getElementById('game-container');

        canvas.width = container.clientWidth * window.devicePixelRatio;

        canvas.height = container.clientHeight * window.devicePixelRatio;

        ctx.scale(window.devicePixelRatio, window.devicePixelRatio);

    }

    window.addEventListener('resize', resizeCanvas);

    resizeCanvas();


    const width = 1024;

    const height = 768;


    // Practice Mode State Machine

    let totalHits = 0;

    let currentStreak = 0;

    let gameState = 'ready'; // ready, pitching, contact, tracking

    let autoPitchTimer = null;

    

    // Smooth Camera Setup

    const cam = { horizon: 310, fov: 380, center: 512 };


    function project3D(x, y, z) {

        const scale = cam.fov / (cam.fov + z);

        return {

            x: cam.center + (x * scale),

            y: cam.horizon + (y * scale),

            scale: scale

        };

    }


    // Ball Simulation Engine

    let ball = { x: 0, y: -45, z: 650, vx: 0, vy: 0, vz: 0, radius: 9, active: false };

    

    // Fluid Batter Geometry Configuration

    let batter = {

        x: -75, y: 170, z: 40,

        batLength: 115,

        swingFrame: 0,

        isSwinging: false,

        angle: -Math.PI / 4

    };


    // visual hitting cage lines tracking 

    let cagePoints = [];

    for(let i=0; i<=650; i+=50) {

        cagePoints.push(i);

    }


    let particles = [];

    function spawnParticles(x, y, color, count=25) {

        for(let i=0; i<count; i++) {

            particles.push({

                x: x, y: y,

                vx: (Math.random() - 0.5) * 18,

                vy: (Math.random() - 0.5) * 18,

                r: Math.random() * 4 + 2,

                alpha: 1,

                color: color

            });

        }

    }


    // Multi-touch & Input configuration

    swingBtn.addEventListener('touchstart', (e) => {

        e.preventDefault();

        executeSwing();

    });

    window.addEventListener('keydown', (e) => {

        if(e.code === 'Space') executeSwing();

    });


    function executeSwing() {

        if (batter.isSwinging) return;

        

        batter.isSwinging = true;

        batter.swingFrame = 1;


        // Dynamic hit engine check

        if (ball.active && gameState === 'pitching') {

            // Check hitting window plane depth

            if (ball.z >= 15 && ball.z <= 95) {

                ball.active = false;

                gameState = 'contact';

                

                const distanceToSweetSpot = Math.abs(ball.z - 50);

                

                if (distanceToSweetSpot < 18) { // Perfect Timing Burst

                    totalHits++;

                    currentStreak++;

                    setFeedback("🔥 CRUSHED!", "fb-great");

                    spawnParticles(project3D(ball.x, ball.y, ball.z).x, project3D(ball.x, ball.y, ball.z).y, '#ff9f43', 35);

                    

                    ball.vx = (Math.random() - 0.5) * 25;

                    ball.vy = -28;

                    ball.vz = 500; 

                } else { // Safe Solid Base Hit

                    totalHits++;

                    currentStreak++;

                    setFeedback("⚾ GOOD HIT!", "fb-good");

                    spawnParticles(project3D(ball.x, ball.y, ball.z).x, project3D(ball.x, ball.y, ball.z).y, '#2ed573', 20);

                    

                    ball.vx = (Math.random() - 0.5) * 18;

                    ball.vy = -14;

                    ball.vz = 300;

                }

                

                updateCounters();

                queueNextPitch(2200);

            }

        }

    }


    function setFeedback(text, className="") {

        pitchFeedback.innerText = text;

        pitchFeedback.className = "hud " + className;

    }


    function updateCounters() {

        hitsVal.innerText = totalHits;

        streakVal.innerText = currentStreak;

    }


    // Unstoppable Batting Practice Pipeline Auto-Throw

    function autoThrowPitch() {

        // Enforce reset on old states safely

        if(autoPitchTimer) clearTimeout(autoPitchTimer);

        

        gameState = 'pitching';

        setFeedback("PITCH COMING...", "");


        // Reset ball path coordinates to launch machine position

        ball.x = (Math.random() - 0.5) * 8; 

        ball.y = -40; 

        ball.z = 650; 

        

        // Balanced easy glide speeds 

        ball.vz = -8.2; 

        ball.vy = 2.0;  

        ball.vx = (Math.random() - 0.5) * 0.4;

        ball.active = true;

    }


    function queueNextPitch(delay) {

        if(autoPitchTimer) clearTimeout(autoPitchTimer);

        gameState = 'tracking';

        autoPitchTimer = setTimeout(() => {

            autoThrowPitch();

        }, delay);

    }


    // Frame Interpolation Pipeline Loop

    function update() {

        // Multi-frame rotational swing script

        if (batter.isSwinging) {

            batter.swingFrame++;

            let progress = batter.swingFrame / 15; // Normalized frame duration

            

            // Complete smooth trigonometric rotational arc

            batter.angle = -Math.PI/4 + (Math.PI * 1.35 * Math.sin(progress * Math.PI / 2));


            if (batter.swingFrame >= 15) {

                batter.isSwinging = false;

                batter.swingFrame = 0;

                batter.angle = -Math.PI / 4; 

            }

        }


        // Processing active pitches towards plate

        if (ball.active && gameState === 'pitching') {

            ball.z += ball.vz;

            ball.y += ball.vy;

            ball.x += ball.vx;


            // Missed execution fallback route

            if (ball.z < -20) {

                ball.active = false;

                currentStreak = 0;

                updateCounters();

                setFeedback("❌ MISS / STRIKE", "fb-miss");

                queueNextPitch(1500); // Instantly queues next automated pitch

            }

        }


        // Forward fly ball physics update engine

        if (!ball.active && (gameState === 'contact' || gameState === 'tracking')) {

            ball.z += ball.vz;

            ball.y += ball.vy;

            ball.x += ball.vx;

            if(ball.y < 350) ball.vy += 0.5; // Simulate smooth gravity drop arc

        }


        // Particle thread execution loops

        for (let i = particles.length - 1; i >= 0; i--) {

            let p = particles[i];

            p.x += p.vx;

            p.y += p.vy;

            p.alpha -= 0.025;

            if (p.alpha <= 0) particles.splice(i, 1);

        }


        // Safe loop structural initialization fallback

        if (gameState === 'ready') {

            queueNextPitch(1000);

        }

    }


    // Dynamic 3D Scene Rendering

    function draw() {

        ctx.clearRect(0, 0, width, height);


        // 1. Render Outfield Green Astro Turf

        ctx.fillStyle = '#10ac84';

        ctx.fillRect(0, cam.horizon, width, height - cam.horizon);


        // 2. Render 3D Batting Tunnel Mesh Guides

        ctx.strokeStyle = 'rgba(255, 255, 255, 0.08)';

        ctx.lineWidth = 2;

        cagePoints.forEach(z => {

            let leftWall = project3D(-180, -120, z);

            let rightWall = project3D(180, -120, z);

            let leftFloor = project3D(-180, 180, z);

            let rightFloor = project3D(180, 180, z);

            

            ctx.beginPath();

            ctx.moveTo(leftWall.x, leftWall.y);

            ctx.lineTo(rightWall.x, rightWall.y);

            ctx.lineTo(rightFloor.x, rightFloor.y);

            ctx.lineTo(leftFloor.x, leftFloor.y);

            ctx.closePath();

            ctx.stroke();

        });


        // 3. Render Solid Clay Infield Batter Box Circle

        let pPlate = project3D(0, 180, 50);

        ctx.fillStyle = '#dd8d54';

        ctx.beginPath();

        ctx.ellipse(pPlate.x, pPlate.y, 140 * pPlate.scale, 50 * pPlate.scale, 0, 0, Math.PI*2);

        ctx.fill();


        // White home plate line draw

        ctx.fillStyle = '#ffffff';

        ctx.beginPath();

        ctx.moveTo(pPlate.x - 22 * pPlate.scale, pPlate.y);

        ctx.lineTo(pPlate.x + 22 * pPlate.scale, pPlate.y);

        ctx.lineTo(pPlate.x + 26 * pPlate.scale, pPlate.y + 16 * pPlate.scale);

        ctx.lineTo(pPlate.x, pPlate.y + 32 * pPlate.scale);

        ctx.lineTo(pPlate.x - 26 * pPlate.scale, pPlate.y + 16 * pPlate.scale);

        ctx.closePath();

        ctx.fill();


        // 4. Clean Mechanical Pitching Machine Object (Replaces weird players)

        let pMachine = project3D(0, 180, 650);

        ctx.fillStyle = '#2c3e50'; // Machine Base housing

        ctx.fillRect(pMachine.x - 16 * pMachine.scale, pMachine.y - 45 * pMachine.scale, 32 * pMachine.scale, 45 * pMachine.scale);

        ctx.fillStyle = '#7f8c8d'; // Launch feeder pipe

        ctx.beginPath();

        ctx.arc(pMachine.x, pMachine.y - 45 * pMachine.scale, 10 * pMachine.scale, 0, Math.PI*2);

        ctx.fill();


        // 5. Render Batter Shadow Geometry

        let pBatter = project3D(batter.x, batter.y, batter.z);

        ctx.fillStyle = 'rgba(0,0,0,0.35)';

        ctx.beginPath();

        ctx.ellipse(pBatter.x, pBatter.y + 10, 42 * pBatter.scale, 14 * pBatter.scale, 0, 0, Math.PI*2);

        ctx.fill();


        // 6. Draw Detailed Batter Character Mesh Model

        ctx.fillStyle = '#0984e3'; // Clean Pro Jersey

        ctx.fillRect(pBatter.x - 18 * pBatter.scale, pBatter.y - 92 * pBatter.scale, 36 * pBatter.scale, 92 * pBatter.scale); 

        ctx.fillStyle = '#ffeedd'; // Skin tone elements

        ctx.beginPath();

        ctx.arc(pBatter.x, pBatter.y - 108 * pBatter.scale, 16 * pBatter.scale, 0, Math.PI*2); 

        ctx.fill();

        ctx.fillStyle = '#222f3e'; // Protection Helmet

        ctx.beginPath();

        ctx.arc(pBatter.x, pBatter.y - 112 * pBatter.scale, 17 * pBatter.scale, Math.PI, 0); 

        ctx.fill();


        // 7. Render 3D Bat with Vector Matrix Rotational sweeps

        ctx.save();

        ctx.translate(pBatter.x + 12 * pBatter.scale, pBatter.y - 68 * pBatter.scale);

        ctx.rotate(batter.angle);

        

        let batGrad = ctx.createLinearGradient(0, 0, batter.batLength * pBatter.scale, 0);

        batGrad.addColorStop(0, '#2d3436'); // Grip tape wrap 

        batGrad.addColorStop(0.2, '#f39c12'); // Natural wood grain barrel

        batGrad.addColorStop(1, '#f1c40f');

        ctx.fillStyle = batGrad;

        

        ctx.beginPath();

        ctx.moveTo(0, -3 * pBatter.scale);

        ctx.lineTo(batter.batLength * pBatter.scale, -8 * pBatter.scale);

        ctx.lineTo(batter.batLength * pBatter.scale, 8 * pBatter.scale);

        ctx.lineTo(0, 3 * pBatter.scale);

        ctx.closePath();

        ctx.fill();

        ctx.restore();


        // 8. Render 3D Ball Asset Profile

        if (ball.active || gameState === 'contact' || gameState === 'tracking') {

            let pBall = project3D(ball.x, ball.y, ball.z);

            let pBallShadow = project3D(ball.x, 180, ball.z);

            

            // Ground track trace shadow circle

            ctx.fillStyle = 'rgba(0,0,0,0.3)';

            ctx.beginPath();

            ctx.arc(pBallShadow.x, pBallShadow.y, (ball.radius * pBallShadow.scale) * 1.1, 0, Math.PI*2);

            ctx.fill();


            // Luminous Baseball Core Render

            ctx.fillStyle = '#ffffff';

            ctx.shadowBlur = 12;

            ctx.shadowColor = '#ffffff';

            ctx.beginPath();

            ctx.arc(pBall.x, pBall.y, ball.radius * pBall.scale, 0, Math.PI*2);

            ctx.fill();

            ctx.shadowBlur = 0; 

        }


        // 9. Particles rendering loop array

        particles.forEach(p => {

            ctx.save();

            ctx.globalAlpha = p.alpha;

            ctx.fillStyle = p.color;

            ctx.beginPath();

            ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);

            ctx.fill();

            ctx.restore();

        });

    }


    function gameLoop() {

        update();

        draw();

        requestAnimationFrame(gameLoop);

    }


    gameLoop();

</script>
